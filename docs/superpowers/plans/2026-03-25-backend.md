> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an Express/PostgreSQL/Prisma backend, migrate the frontend from hardcoded data to API-driven rendering, and add context-menu interactions for adding/removing nodes.

**Architecture:** Thin REST API (Express + Prisma) with a single read endpoint and three mutation endpoints. Frontend re-fetches and fully re-renders after every mutation. Express also serves `index.html` as a static file so one process handles everything.

**Tech Stack:** Node.js, Express 4, PostgreSQL, Prisma 5, Jest, Supertest, D3.js v7 (existing)

```prisma
datasource db {
  provider = "postgresql"

model Person {
  id        String        @id @default(cuid())
  name      String
  birth     Int
  death     Int?
  gender    String
  spouseAIn Couple?       @relation("SpouseA")
  spouseBIn Couple?       @relation("SpouseB")
  childIn   CoupleChild[]
}

model Couple {
  id        String        @id @default(cuid())
  spouseA   Person        @relation("SpouseA", fields: [spouseAId], references: [id])
  spouseAId String        @unique
  spouseB   Person        @relation("SpouseB", fields: [spouseBId], references: [id])
  spouseBId String        @unique
  children  CoupleChild[]
}

model CoupleChild {
  couple    Couple  @relation(fields: [coupleId], references: [id], onDelete: Cascade)
  coupleId  String
  child     Person  @relation(fields: [childId], references: [id])
  childId   String
  sortOrder Int     @default(0)

  @@id([coupleId, childId])
}
```

```js
function errorHandler(err, req, res, next) {
  if (err.status) {
    return res.status(err.status).json({ error: err.message });
  }
  if (err.code === 'P2002') {
    return res.status(409).json({ error: 'Unique constraint violation' });
  }
  if (err.code === 'P2025') {
    return res.status(404).json({ error: 'Record not found' });
  }
  console.error(err);
  return res.status(500).json({ error: 'Internal server error' });
}

module.exports = errorHandler;
```

- [ ] **Step 4: Create `server/src/lib/createPerson.js`**

```js
function createPerson(data, tx) {
  const { name, birth, death, gender } = data;

  if (!name || typeof name !== 'string' || name.trim() === '') {
    const err = new Error('name is required');
    err.status = 400;
    throw err;
  }
  if (!Number.isInteger(birth) || birth < 1000 || birth > 2100) {
    const err = new Error('birth must be an integer between 1000 and 2100');
    err.status = 400;
    throw err;
  }
  if (death !== undefined && death !== null) {
    if (!Number.isInteger(death) || death < birth) {
      const err = new Error('death must be an integer >= birth year');
      err.status = 400;
      throw err;
    }
  }
  if (gender !== 'M' && gender !== 'F') {
    const err = new Error('gender must be "M" or "F"');
    err.status = 400;
    throw err;
  }

  return tx.person.create({
    data: { name: name.trim(), birth, death: death ?? null, gender },
  });
}

module.exports = { createPerson };
```

- [ ] **Step 5: Create `server/src/routes/tree.js`**

```js
const { Router } = require('express');
const prisma = require('../lib/prisma');

const router = Router();

router.get('/tree', async (req, res, next) => {
  try {
    const [people, couplesRaw] = await Promise.all([
      prisma.person.findMany(),
      prisma.couple.findMany({
        include: { children: { orderBy: { sortOrder: 'asc' } } },
      }),
    ]);

    // Sort so the root couple (neither spouse is a child anywhere) comes first
    const childIds = new Set(couplesRaw.flatMap(c => c.children.map(cc => cc.childId)));
    const rootIdx = couplesRaw.findIndex(
      c => !childIds.has(c.spouseAId) && !childIds.has(c.spouseBId)
    );
    const sorted = rootIdx > 0
      ? [couplesRaw[rootIdx], ...couplesRaw.filter((_, i) => i !== rootIdx)]
      : couplesRaw;

    const couples = sorted.map(c => ({
      id: c.id,
      spouseA: c.spouseAId,
      spouseB: c.spouseBId,
      children: c.children.map(cc => cc.childId),
    }));

    res.json({ people, couples });
  } catch (err) {
    next(err);
  }
});

module.exports = router;
```

```js
const prisma = require('../lib/prisma');

const router = Router();

router.delete('/:id', async (req, res, next) => {
  try {
    const { id } = req.params;

    const person = await prisma.person.findUnique({ where: { id } });
    if (!person) return res.status(404).json({ error: 'Person not found' });

    const couple = await prisma.couple.findFirst({
      where: { OR: [{ spouseAId: id }, { spouseBId: id }] },
      include: { children: true },
    });

    if (couple && couple.children.length > 0) {
      return res.status(409).json({ error: 'Cannot remove a person who has children' });
    }

    const deleted = [id];

    await prisma.$transaction(async tx => {
      if (couple) {
        if (couple.spouseAId === id) {
          // Also delete spouseB (would become unreachable)
          const spouseBId = couple.spouseBId;
          // Remove spouseB from any parent couple's children list (defensive)
          await tx.coupleChild.deleteMany({ where: { childId: spouseBId } });
          // Delete the couple (cascade removes its own CoupleChild rows)
          await tx.couple.delete({ where: { id: couple.id } });
          // Remove target (spouseA) from their parent couple's children list
          await tx.coupleChild.deleteMany({ where: { childId: id } });
          // Delete spouseB person
          await tx.person.delete({ where: { id: spouseBId } });
          deleted.push(spouseBId);
        } else {
          // Deleting spouseB — dissolve couple, spouseA remains
          await tx.couple.delete({ where: { id: couple.id } });
          await tx.coupleChild.deleteMany({ where: { childId: id } });
        }
      } else {
        await tx.coupleChild.deleteMany({ where: { childId: id } });
      }
      await tx.person.delete({ where: { id } });
    });

    res.json({ deleted });
  } catch (err) {
    next(err);
  }
});
```

```js
  ctxConfirmArea.innerHTML = '';

  // Determine state
  const personId = d.person.id;
  const myCouple = couples.find(c => c.spouseA === personId || c.spouseB === personId);
  const hasCouple = !!myCouple;
  const hasChildren = hasCouple && myCouple.children.length > 0;
  const isSpouseA = hasCouple && myCouple.spouseA === personId;

  // Add Spouse — only if no couple
  ctxAddSpouse.disabled = hasCouple;
  ctxAddSpouse.title = hasCouple ? 'Already has a spouse' : '';

  // Add Child — only if in a couple
  ctxAddChild.disabled = !hasCouple;
  ctxAddChild.title = !hasCouple ? 'Add a spouse first' : '';

  // Remove — disabled if couple has children
  ctxRemove.disabled = hasChildren;
  ctxRemove.title = hasChildren ? 'Cannot remove a person with children' : '';

  // Position menu
  const x = Math.min(event.clientX, window.innerWidth  - 200);
  const y = Math.min(event.clientY, window.innerHeight - 160);
  ctxMenu.style.left    = x + 'px';
  ctxMenu.style.top     = y + 'px';
  ctxMenu.style.display = 'block';
}

ctxAddSpouse.addEventListener('click', () => {
  if (!ctxTarget) return;
  openModal('spouse', ctxTarget);
  hideCtxMenu();
});

ctxAddChild.addEventListener('click', () => {
  if (!ctxTarget) return;
  openModal('child', ctxTarget);
  hideCtxMenu();
});

ctxRemove.addEventListener('click', () => {
  if (!ctxTarget || ctxRemove.disabled) return;
  const d = ctxTarget;
  const personId = d.person.id;
  const myCouple = couples.find(c => c.spouseA === personId || c.spouseB === personId);
  const isSpouseA = myCouple && myCouple.spouseA === personId;
  const spouseBName = isSpouseA ? personMap[myCouple.spouseB]?.name : null;

  let msg = `Remove ${d.person.name}?`;
  if (isSpouseA && spouseBName) msg += ` This will also remove their spouse ${spouseBName}.`;
  else if (myCouple) msg += ` Their couple will be dissolved.`;
  msg += ' This cannot be undone.';

  ctxConfirmArea.innerHTML = `
    <div class="ctx-confirm">
      <div>${msg}</div>
      <button class="btn-danger" id="ctx-confirm-yes">Confirm</button>
      <button class="btn-cancel" id="ctx-confirm-no">Cancel</button>
    </div>`;

  document.getElementById('ctx-confirm-no').addEventListener('click', hideCtxMenu);
  document.getElementById('ctx-confirm-yes').addEventListener('click', async () => {
    ctxInflight = true;
    ctxConfirmArea.innerHTML = '<div class="ctx-confirm">Removing…</div>';
    try {
      const res = await fetch(`/api/people/${personId}`, { method: 'DELETE' });
      const body = await res.json();
      if (!res.ok) throw new Error(body.error || 'Request failed');
      ctxInflight = false;
      hideCtxMenu();
      await init();
    } catch (err) {
      ctxInflight = false;
      ctxConfirmArea.innerHTML = `
        <div class="ctx-confirm">
          <div class="ctx-error">${err.message}</div>
          <button class="btn-cancel" id="ctx-confirm-no">Close</button>
        </div>`;
      document.getElementById('ctx-confirm-no').addEventListener('click', hideCtxMenu);
    }
  });
});
```

```js
// ===== MODAL =====
const modalOverlay = document.getElementById('modal-overlay');
const modalTitle   = document.getElementById('modal-title');
const modalName    = document.getElementById('modal-name');
const modalBirth   = document.getElementById('modal-birth');
const modalDeath   = document.getElementById('modal-death');
const modalError   = document.getElementById('modal-error');
const modalSubmit  = document.getElementById('modal-submit');
let modalContext   = null; // { mode: 'spouse'|'child', target: d }

function openModal(mode, d) {
  modalContext = { mode, target: d };
  modalTitle.textContent = mode === 'spouse' ? 'Add Spouse' : 'Add Child';
  modalName.value  = '';
  modalBirth.value = '';
  modalDeath.value = '';
  document.querySelectorAll('input[name="modal-gender"]').forEach(r => r.checked = false);
  modalError.textContent = '';
  modalOverlay.classList.add('open');
  modalName.focus();
}

function closeModal() {
  modalOverlay.classList.remove('open');
  modalContext = null;
}

document.getElementById('modal-cancel').addEventListener('click', closeModal);
modalOverlay.addEventListener('click', e => { if (e.target === modalOverlay) closeModal(); });

modalSubmit.addEventListener('click', async () => {
  if (!modalContext) return;
  modalError.textContent = '';

  const name   = modalName.value.trim();
  const birth  = parseInt(modalBirth.value, 10);
  const death  = modalDeath.value ? parseInt(modalDeath.value, 10) : null;
  const gender = document.querySelector('input[name="modal-gender"]:checked')?.value;

  if (!name)            { modalError.textContent = 'Name is required.'; return; }
  if (!birth)           { modalError.textContent = 'Birth year is required.'; return; }
  if (birth < 1000 || birth > 2100) { modalError.textContent = 'Birth year must be between 1000 and 2100.'; return; }
  if (death !== null && death < birth) { modalError.textContent = 'Death year must be ≥ birth year.'; return; }
  if (!gender)          { modalError.textContent = 'Gender is required.'; return; }

  const personData = { name, birth, death, gender };
  const { mode, target } = modalContext;

  try {
    modalSubmit.disabled = true;
    let res;
    if (mode === 'spouse') {
      res = await fetch('/api/couples', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ existingPersonId: target.person.id, spouse: personData }),
      });
    } else {
      const coupleId = target.coupleId;
      res = await fetch(`/api/couples/${coupleId}/children`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(personData),
      });
    }

    const body = await res.json();
    if (!res.ok) throw new Error(body.error || 'Request failed');

    closeModal();
    await init();
  } catch (err) {
    modalError.textContent = err.message;
  } finally {
    modalSubmit.disabled = false;
  }
});
```
