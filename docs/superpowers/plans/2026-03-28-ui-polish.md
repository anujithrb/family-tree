- [ ] **Step 1: Add `splitName()` just before `renderNodes()`**

Find the line `function renderNodes() {` (~line 573). Insert the following function immediately above it:

```js
    // Splits a person's name into at most two lines that fit within the 70px
    // text area (x=46 to x=116). Estimates character width at 7px for 12px/600
    // system-ui — a conservative proxy that avoids needing a hidden canvas.
    function splitName(name) {
      const MAX_W = 70;
      const CHAR_W = 7;
      const words = name.split(' ');
      let line1 = '';
      let i = 0;
      while (i < words.length) {
        const candidate = line1 ? line1 + ' ' + words[i] : words[i];
        // Accept if it fits, or force at least one word onto line 1
        if (candidate.length * CHAR_W <= MAX_W || !line1) {
          line1 = candidate;
          i++;
        } else {
          break;
        }
      }
      if (i >= words.length) return { line1, line2: null };
      let line2 = words.slice(i).join(' ');
      if (line2.length * CHAR_W > MAX_W) {
        const max = Math.floor(MAX_W / CHAR_W) - 1; // leave room for "…"
        line2 = line2.slice(0, max) + '\u2026';
      }
      return { line1, line2 };
    }

```
